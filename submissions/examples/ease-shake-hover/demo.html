Usage

<!-- Basic shake on hover -->
<button class="ease-shake-hover">Shake Me!</button>
<div class="ease-shake-hover">Any element shakes on hover</div>

<!-- Different intensities -->
<div class="ease-shake-subtle">Subtle shake</div>
<div class="ease-shake-strong">Strong shake</div>

<!-- Different directions -->
<div class="ease-shake-horizontal">Left-right shake</div>
<div class="ease-shake-vertical">Up-down shake</div>
<div class="ease-shake-diagonal">Diagonal shake</div>

<!-- Different speeds -->
<div class="ease-shake-fast">Fast shake</div>
<div class="ease-shake-slow">Slow shake</div>
