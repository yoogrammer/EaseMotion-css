CSS.escape Feature Detection

Calling CSS.escape() without a feature detection guard can throw in older browsers.

Before: No guard

Calls CSS.escape(id) directly. Throws TypeError if CSS.escape is undefined.

function openModal(id) {
  // UNSAFE: no feature detection
  const safeId = CSS.escape(id);
  const el = document.getElementById(safeId);
  el.classList.add('is-open');
}
⚠ TypeError if CSS or CSS.escape is not defined

After: With guard

Feature-detects CSS.escape and falls back to a manual escape function.

function safeEscape(id) {
  if (typeof CSS !== 'undefined'
      && typeof CSS.escape === 'function') {
    return CSS.escape(id);
  }
  // Manual fallback
  return id.replace(
    /[!"#$%&'()*+,./:;<=>?@[\]^`{|}~]/g,
    '\\$&'
  );
}

function openModal(id) {
  const safeId = safeEscape(id);
  const el = document.getElementById(safeId);
  if (el) el.classList.add('is-open');
}
✔ Graceful fallback — never throws

Recommended pattern

Always guard CSS.escape() with a feature detection check and provide a manual fallback for older browsers:

function safeCSSEscape(value) {
  const hasCSSEscape = typeof CSS !== 'undefined'
    && typeof CSS.escape === 'function';

  if (hasCSSEscape) return CSS.escape(value);

  // Fallback: escape CSS special characters
  return String(value).replace(
    /[!"#$%&'()*+,.\/:;<=>?@[\\\]^`{|}~]/g,
    '\\$&'
  );
}
