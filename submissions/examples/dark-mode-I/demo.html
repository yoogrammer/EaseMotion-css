Form Elements

Implementation

/* Light mode variables (default) */
:root {
    --bg-primary: #ffffff;
    --bg-secondary: #f8fafc;
    --text-primary: #1e293b;
    --text-secondary: #64748b;
    --primary-color: #667eea;
    --border-color: #e2e8f0;
}

/* Dark mode overrides */
@media (prefers-color-scheme: dark) {
    :root {
        --bg-primary: #0f172a;
        --bg-secondary: #1e293b;
        --text-primary: #f1f5f9;
        --text-secondary: #94a3b8;
        --primary-color: #818cf8;
        --border-color: #334155;
    }
}
