How to Use

html
<!-- Basic: children slide up one by one on scroll -->
<div class="stagger-group" data-stagger="100">
  <div>Item 1</div>
  <div>Item 2</div>
  <div>Item 3</div>
</div>

<!-- Custom animation -->
<div class="stagger-group" data-stagger="150" data-animation="slide-left">
  <div>Item 1</div>
  <div>Item 2</div>
</div>

<!-- Available animations: slide-up, slide-left, fade, zoom -->
