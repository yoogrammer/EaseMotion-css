Usage

<!-- Basic usage -->
<div class="ease-word-by-word">Your text here</div>

<!-- Different speeds -->
<div class="ease-word-by-word-fast">Fast animation</div>
<div class="ease-word-by-word-slow">Slow animation</div>

<!-- Different directions -->
<div class="ease-word-by-word-fade">Fade only</div>
<div class="ease-word-by-word-left">Slide from left</div>

<!-- Custom delay -->
<div class="ease-word-by-word" style="--ease-word-delay: 150ms;">
    Custom delay
</div>
