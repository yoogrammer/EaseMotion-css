Usage

<!-- HTML -->
<div class="tilt-card" data-tilt-max="15">
    <div class="tilt-content">
        <h3>Card Title</h3>
        <p>Card content</p>
    </div>
</div>

<!-- JavaScript -->
document.querySelectorAll('.tilt-card').forEach(card => {
    const maxTilt = card.getAttribute('data-tilt-max') || 15;
    card.addEventListener('mousemove', (e) => {
        const rect = card.getBoundingClientRect();
        const x = e.clientX - rect.left;
        const y = e.clientY - rect.top;
        const rotateY = ((x - rect.width/2) / (rect.width/2)) * maxTilt;
        const rotateX = ((rect.height/2 - y) / (rect.height/2)) * maxTilt;
        card.style.transform = `perspective(1000px) rotateX(${rotateX}deg) rotateY(${rotateY}deg)`;
    });
    card.addEventListener('mouseleave', () => {
        card.style.transform = 'perspective(1000px) rotateX(0deg) rotateY(0deg)';
    });
});

⚡ Parallax Tilt Features

  • 🌀 3D tilt based on cursor position
  • 📐 Adjustable intensity via data-tilt-max
  • ✨ Smooth reset on mouse leave
  • 💫 Pure CSS + lightweight JavaScript
  • 🎨 Works on cards, images, and any element
