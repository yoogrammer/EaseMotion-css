How to Use

html
<!-- Wrap cards in a perspective container -->
<div class="ease-bento-container">
  <div class="ease-bento-card">
    <!-- Background mesh, stays far back -->
    <div class="ease-bento-bg" aria-hidden="true"></div>

    <!-- Content floats forward -->
    <div class="ease-bento-content">
      <div class="ease-bento-icon">📊</div>
      <h3>Analytics Engine</h3>
      <p>Multi-layered parallax depth interaction.</p>
    </div>
  </div>
</div>

<!-- Background color variants -->
<div class="ease-bento-bg ease-bento-bg--green"></div>
<div class="ease-bento-bg ease-bento-bg--pink"></div>
<div class="ease-bento-bg ease-bento-bg--amber"></div>

<!-- Grid sizing modifiers -->
<div class="ease-bento-card bento-large">...</div>
<div class="ease-bento-card bento-wide">...</div>
