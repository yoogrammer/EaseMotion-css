Section 05

How to Use

Add the class to fire once. Remove and re-add to replay — the JS snippet below handles this with a one-line reflow trick.

<!-- One-shot: add class to fire --> <button class="ease-tada">✓ Success</button> <!-- Hover: no JS needed --> <button class="ease-tada-hover">🚀 Get Started</button> <!-- Loop: repeats every ~3s --> <span class="badge ease-tada-loop">3</span> <!-- Speed presets --> <button class="ease-tada ease-tada-fast">⚡ Fast</button> <button class="ease-tada ease-tada-slow">🎭 Slow</button> <!-- Custom duration --> <button class="ease-tada" style="--ease-tada-duration: 0.8s;">Custom</button> /* Replay via JS — remove, reflow, re-add */ function triggerTada(el) { el.classList.remove('ease-tada'); void el.offsetWidth; // force reflow el.classList.add('ease-tada'); }
