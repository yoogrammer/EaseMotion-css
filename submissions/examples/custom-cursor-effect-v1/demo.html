Usage

<!-- 1. Add cursor elements to <body> -->
<div class="em-cursor"></div>
<div class="em-cursor-ring"></div>
 
<!-- 2. Link the stylesheet -->
<link rel="stylesheet" href="style.css">
 
<!-- 3. (Optional) Override the theme color -->
<style>
  :root { --em-cursor-color: #00d2a8; }
</style>
 
<!-- 4. Add the JS snippet (cursor.js or inline) -->
<script src="cursor.js"></script>
 
<!-- 5. Mark magnetic elements with data-magnetic -->
<button data-magnetic>Click me</button>
