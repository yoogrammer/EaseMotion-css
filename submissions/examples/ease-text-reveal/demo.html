Usage

<!-- Basic text reveal -->
<div class="ease-text-reveal">
    <span>Your text here</span>
</div>

<!-- With custom duration -->
<div class="ease-text-reveal" style="--ease-reveal-duration: 1.5s;">
    <span>Slow reveal</span>
</div>

<!-- With delay -->
<div class="ease-text-reveal" style="--ease-reveal-delay: 0.5s;">
    <span>Delayed reveal</span>
</div>

<!-- Word by word reveal -->
<div class="ease-text-reveal-words">
    <span>Word</span> <span>by</span> <span>word</span>
</div>
