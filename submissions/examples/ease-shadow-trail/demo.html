Usage

<!-- Basic shadow trail -->
<div class="ease-shadow-trail">
    Content with shadow trail
</div>

<!-- Custom shadow color -->
<div class="ease-shadow-trail" style="--ease-shadow-color: rgba(255,255,255,0.3);">
    Custom color
</div>

<!-- Custom depth -->
<div class="ease-shadow-trail" style="--ease-shadow-depth: 20px;">
    Deeper trail
</div>
