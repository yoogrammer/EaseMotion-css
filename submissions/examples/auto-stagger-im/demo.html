How to Use

html
<!-- Just add class + optional speed variable -->
<div class="auto-stagger" style="--stagger-delay: 80ms">
  <div>Item 1</div>   <!-- delay: 0ms  -->
  <div>Item 2</div>   <!-- delay: 80ms -->
  <div>Item 3</div>   <!-- delay: 160ms -->
  <div>Item 4</div>   <!-- delay: 240ms -->
</div>

<!-- Animation variants -->
<div class="auto-stagger auto-stagger--fade">...</div>
<div class="auto-stagger auto-stagger--zoom">...</div>
<div class="auto-stagger auto-stagger--slide-left">...</div>
