Backend Implementation Reference

Nodemailer Utility

This utility is called inside `adminController.updateApplicationStatus` to send customized HTML status letters. The credentials `EMAIL_USER` and `EMAIL_PASS` are read dynamically from environment variables.

backend/utils/emailService.js 
import nodemailer from 'nodemailer';

// Configure SMTP transporter
const transporter = nodemailer.createTransport({
  service: 'gmail',
  auth: {
    user: process.env.EMAIL_USER,
    pass: process.env.EMAIL_PASS
  }
});

/**
 * Sends a stylized HTML application status notification to students
 */
export const sendStatusEmail = async (studentEmail, studentName, jobTitle, company, status) => {
  const isShortlisted = status === 'shortlisted';
  
  const mailOptions = {
    from: `"Placement Portal" <${process.env.EMAIL_USER}>`,
    to: studentEmail,
    subject: `Application Update: ${jobTitle} at ${company}`,
    html: `
      <div style="font-family: sans-serif; padding: 20px; border: 1px solid #e2e8f0; border-radius: 8px; max-width: 600px; margin: 0 auto;">
        <h2 style="color: #1e293b;">Hello ${studentName},</h2>
        <p style="color: #475569;">There is an update on your application for <strong>${jobTitle}</strong> at <strong>${company}</strong>.</p>
        <div style="padding: 15px; margin: 15px 0; border-radius: 6px; font-weight: bold; text-align: center; font-size: 18px; 
                    background-color: ${isShortlisted ? '#d1fae5' : '#fee2e2'}; 
                    color: ${isShortlisted ? '#065f46' : '#991b1b'};">
          STATUS: ${status.toUpperCase()}
        </div>
        ${isShortlisted 
          ? '<p style="color: #475569;">Congratulations! Please check the dashboard regularly for your online test/interview links and details.</p>'
          : '<p style="color: #475569;">Thank you for your effort. We encourage you to apply for other matching positions on the portal.</p>'}
        <br/>
        <p style="color: #475569;">Best Regards,</p>
        <p style="color: #1e293b;"><strong>Training & Placement Cell</strong></p>
      </div>
    `
  };

  return transporter.sendMail(mailOptions);
};
