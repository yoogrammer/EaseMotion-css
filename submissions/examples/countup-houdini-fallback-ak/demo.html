How the fallback works

Detect @property support using CSS.supports() or by trying to register a property. If unsupported, fall back to a JS-driven animation loop.

if (CSS.supports('--num: 0')) {
  // Use CSS @property animation
} else {
  // Use requestAnimationFrame fallback
  let val = 0;
  function step() {
    if (val >= 100) return;
    val += 2;
    el.textContent = val;
    requestAnimationFrame(step);
  }
  requestAnimationFrame(step);
}
